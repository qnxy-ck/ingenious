namespace com.demo.mapper

import entity com.demo.data.entity.UserInfo


// 对字段调用函数返回处理后的值
search selectByUsername(username: String): multi {

    select
        id,
        username,
        phone_num,       => this::phoneNumFormat()   // this表示当前字段的值, 调用函数phoneNumFormat进行格式化参数返回 199****9999 格式
        age,
        birthday,
        created_at
    from user_info
    where username = :username?
    
}
