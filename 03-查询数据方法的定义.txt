namespace org.ingenious.demo.mapper

import entity org.ingenious.demo.entity.UserInfo

import java.util.List

// search 为查询方法定义关键字
// selectById 该方法的名称
// id: Long 为查询参数
// UserInfo 查询映射类型. 如果该类型和 entity 类型一致, 则可省略
search selectById(id: Long): UserInfo {
    // sql
}


// multi 返回多个数据
// UserInfo 查询映射类型. 如果该类型和 entity 类型一致, 则可省略
search selectByName(name: String): multi UserInfo {
    // sql
}


// idList: multi Long
// multi 对参数使用 标识参数为多个
search selectByIds(idList: multi Long): multi {
    // sql
}


// 新增单参数查询声明简化方式
// 单参数必须是实体类中存在的字段

// 对应声明为: search selectById(id: Long)
search #select_by :id

// 对应声明为: search selectByUsername(username: String): multi
search multi #select_by :username


// 对应声明: search selectPageByUsername(username: String, pageInfo: PageInfo): page
search page #select_by :username
